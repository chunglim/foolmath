\begin{alignat*}{5}
&\,&e^{i\pi}\quad&=\quad&&-1\\
&\,&\ln{e^{i\pi}}\quad&=&&\ln{(-1)}\\
&\,&i\pi\cancelto{1}{\ln{e}}\quad&=&&\ln{(-1)}\\
&\,&1^{i\pi}\quad&=&&1^{\ln{(-1)}}\\
&\,&(1^\pi)^i\quad&=&&1^{\ln{(-1)}}\\
&\,&1^i\quad&=&&1^{\ln{(-1)}}\\
&\,&\log_{1}1^i\quad&=&&\log_{1}1^{\ln{(-1)}}\\
&\,&i\log_{1}1\quad&=&&\ln{(-1)}\log_{1}1\\
&\,&i\quad&=&&\ln{(-1)}\\
&\,&i\quad&=&&\ln{(e^{i\pi})}\\
&\,&\cancel{i}\quad&=&&\cancel{i}\pi\cancelto{1}{\ln{e}}\\
&\small\text{Thus}&\pi\quad&=&&1
\end{alignat*}
