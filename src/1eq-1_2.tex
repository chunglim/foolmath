\begin{alignat*}{5}
&\,&&\quad\frac{-1}{1}\quad&&\,=\quad\frac{1}{-1}\\
&\,&&\sqrt{\frac{-1}{1}}\quad&&\,=\quad\sqrt{\frac{1}{-1}}\\
&\,&&\quad\frac{i}{1}\quad&&\,=\quad\frac{1}{i}\\
&\,&&\quad i\quad&&\,=\quad\frac{1}{i}\\
&\,&&\quad i^2\quad&&\,=\quad1\\
&\text{Thus}\quad&&\quad-1\quad&&\,=\quad1
\end{alignat*}
