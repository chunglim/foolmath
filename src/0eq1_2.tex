&\,&\int\frac{1}{x}\,dx\quad&=&&\cancel{x}\frac{1}{\cancel{x}}-\int x\,d\left(\frac{1}{x}\right)\\
&\,&\,&=&&1-\int x\,d(x^{-1})\\
&\,&\,&=&&1-\int x\,(-1x^{-2})\,dx\\
&\,&\,&=&&1+\int x.x^{-2}\,dx\\
&\,&\cancel{\int\frac{1}{x}\,dx}\quad&=&&1+\cancel{\int\frac{1}{x}\,dx}\\
&\small\text{Thus}\normalsize\qquad&0\quad&=&&1
\end{alignat*}
