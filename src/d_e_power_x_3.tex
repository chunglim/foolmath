\begin{alignat*}{5}
&\,&\frac{dy}{dx}\quad&=\quad&&y\\
&\,&\frac{dy}{y}\quad&=&&dx\\
&\small\text{integral}\quad&\int\frac{dy}{y}\quad&=&&\int{dx}\\
&\,&\ln{y}\quad&=&&x\qquad&\small\text{omit}~C\\
&\,&y\quad&=&&e^x\\
&\small\text{Thus}&\frac{d}{dx}e^x\quad&=&&e^x\qquad&\small\text{(valid proof)}
\end{alignat*}
