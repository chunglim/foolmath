\begin{alignat*}{5}
&\,&\frac{1}{i}\quad&=\quad&&i^{-1}\\
&\,&\,&=&&\sqrt{-1}^{-1}\\
&\,&\,&=&&\left((-1)^\frac{1}{2}\right)^{-1}\\
&\,&\,&=&&\left((-1)^{-1}\right)^\frac{1}{2}\\
&\,&\,&=&&\left(\frac{1}{-1}\right)^\frac{1}{2}\\
&\,&\,&=&&-1^\frac{1}{2}\\
&\,&\,&=&&\sqrt{-1}\\
&\small\text{Thus}\normalsize\qquad&\frac{1}{i}\quad&=&& i\qquad\small\text{(fool proof)}
\end{alignat*}
