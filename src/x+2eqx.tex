\begin{alignat*}{5}
&\,&x+2\quad&=\quad&&x\qquad&\small\text{(1)}\\
&\,&(x+2)^2\quad&=&& x^2\\
&\,&\cancel{x^2}+4x+4\quad&=&&\cancel{x^2}\\
&\,&4x\quad&=&&-4\\
&\,&x\quad&=&&-1\qquad&\small\text{(2)}\\
&\small\text{substitute (2) in (1)}\qquad&-1+2\quad&=&&-1\\
&\small\text{Thus}&1\quad&=&&-1
\end{alignat*}
