\begin{alignat*}{5}
&\,&1\quad&=\quad&&1\\
&\,&\,&=&&1\cdot1\\
&\,&\,&=&&1\cdot(-1)^2\\
&\,&\,&=&&1\cdot(i^2)^2\\
&\,&\,&=&&1\cdot i^4\\
&\small\sqrt{}\text{ both sides}\qquad&\sqrt{1}\quad&=&&\sqrt{1\cdot i^4}\\
&\,&1\quad&=&&\sqrt{1}\cdot\sqrt{i^4}\\
&\,&\,&=&&1\cdot\left(i^4\right)^\frac{1}{2}\\
&\,&\,&=&&1\cdot i^2\\
&\,&\,&=&&1\cdot-1\\
&\qquad\small\text{Thus}&1\quad&=&&-1
\end{alignat*}
