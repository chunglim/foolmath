\begin{alignat*}{5}
&\,&&e^i\quad&&\,=\quad\left(e^i\right)^\frac{2\pi}{2\pi}\\
&\,&&\quad&&\,=\quad\left(e^{i\pi}\right)^{2.\frac{\pi}{2\pi}}\\
&\,&&\quad&&\,=\quad(-1)^{2.\frac{\cancel{\pi}}{2\cancel{\pi}}}\\
&\,&&\quad&&\,=\quad1^\frac{1}{2}\\
&\text{Thus}\quad&&e^i\quad&&\,=\quad1
\end{alignat*}
