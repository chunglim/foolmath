\begin{alignat*}{3}
&\,&&\,\quad e^{i\pi}&&\,=\qquad-1\\
&\,&&\,\ln{e^{i\pi}}&&\,=\qquad\ln{(-1)}\\
&\,&&\,i\pi\cancelto{1}{\ln e}&&\,=\qquad\ln{(-1)}\\
&\,&&\,\quad\pi&&\,=\qquad\frac{\ln{(-1)}}{i}\\
&\,&&\,&&\,=\qquad\frac{2\ln{(-1)}}{2i}\\
&\,&&\,&&\,=\qquad\frac{\ln{(-1)^2}}{2i}\\
&\,&&\,&&\,=\qquad\frac{\cancelto{0}{\ln{1}}}{2i}\\
&\mathrm{Thus}\qquad&&\,\quad\pi&&\,=\qquad0
\end{alignat*}
