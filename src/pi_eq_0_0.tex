\begin{alignat*}{5}
&\,&e^{i\pi}\quad&=\quad&&-1\\
&\,&\ln{e^{i\pi}}\quad&=&&\ln{(-1)}\\
&\,&i\pi\cancelto{1}{\ln e}\quad&=&&\ln{(-1)}\\
&\,&\pi\quad&=&&\frac{\ln{(-1)}}{i}\\
&\,&\,&=&&\frac{2\ln{(-1)}}{2i}\\
&\,&\,&=&&\frac{\ln{(-1)^2}}{2i}\\
&\,&\,&=&&\frac{\cancelto{0}{\ln{1}}}{2i}\\
&\small\text{Thus}&\pi\quad&=&&0
\end{alignat*}
