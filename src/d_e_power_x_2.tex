\begin{alignat*}{5}
&\,&y\quad&=\quad&&e^x\\
&\,&\ln{y}\quad&=&&\ln{e^x}\\
&\,&\,&=&&x\cancelto{1}{\ln{e}}\\
&\,&\ln{y}\quad&=&&x\\
&\small\text{diff both sides}\normalsize\quad&\frac{d}{dx}\ln{y}\quad&=&&\frac{dx}{dx}\\
&\,&\frac{1}{y}\frac{dy}{dx}\quad&=&&1\\
&\,&\frac{dy}{dx}\quad&=&&y\\
&\qquad\small\text{Thus}\normalsize\quad&\frac{d}{dx}e^x\quad&=&&e^x\quad\small\text{(valid proof)}
\end{alignat*}
